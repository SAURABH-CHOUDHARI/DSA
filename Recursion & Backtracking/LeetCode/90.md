### **Optimized Solution (Sorting + Backtracking)**
```js
var subsetsWithDup = function(nums) {
    nums.sort();
    let ans = [];
    
    function solve(nums, index, current) {
        ans.push([...current]);
        
        for (let i = index; i < nums.length; i++) {
           
            if (i > index && nums[i] === nums[i-1]) continue;
            
            current.push(nums[i]);
            solve(nums, i + 1, current);
            current.pop();
        }
    }
    
    solve(nums, 0, []);
    return ans;
};
```
