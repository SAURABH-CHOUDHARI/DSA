1. **Edge Case**: If `n == 1`, return `true` since `2^0 = 1`.

    if (n == 1) return true;

    if (!(n & 1)) { // Check if n is even
        let res = 2;
        while (res < n) {
            res *= 2;
        }

    }
    return false;
};
