```go
func lengthOfLongestSubstring(s string) int {
    charSet := make(map[rune]int)
    maxLen := 0
    left := 0

    for right, char := range s {
        if idx, ok := charSet[char]; ok && idx >= left {
            left = idx + 1
        }
        maxLen = max(maxLen, right-left+1)
        charSet[char] = right
    }

    return maxLen
}

func max(a, b int) int {
    if a > b {
        return a
    }
    return b
}
```
